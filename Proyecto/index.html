Pseudocódigo
Funcion entero binarySearch(list lalista,entero dato)

D.V.
Definir menor como entero
Definir mayor como entero
Definir centro como entero
Inicio
menor <- 0
mayor <- tamaño(lalista) - 1
Mientras (menor <= mayor)
  centro <- (menor + mayor)/2
FinMientras
Si (lalista[centro] == dato) entonces
  Devolver centro
SiNo
  Si(lalista[centro] < dato) entonces
   menor <- centro + 1
  SiNo
   mayor <- centro - 1
  FinSi
FinSi
Devolver -1
FinFuncion


Pseudocódigo
Funcion entero linearSearch(list lalista,entero dato)

D.V.
Definir i como entero
Inicio
i <- 0
Mientras (i < tamaño(lalista) y lalista[i] < dato)
  i <- i+1
FinMientras
Si (i >= tamaño(lalista) o lalista[i] != dato) entonces
  Devolver -1
FinSi
Devolver i
FinFuncion

Funcion entero quickSelect(list lalista,entero izquierda,entero derecha, entero k)

D.V.
Definir part como entero
Inicio
part <- particion(lalista,izquierda,derecha)
Si (part == k) entonces
  Devolver lalista[part]
SiNo
   Si (particion < k) entonces
   Devolver quickSelect(lalista,part + 1,derecha,k)
  FinSi
SiNo
  Devolver quickSelect(lalista, izquierda,part - 1,k)
FinSi
FinFuncion

Funcion entero particion(list lalista,entero izquierda,entero derecha)
D.V.
Definir i como entero
Definir pivote como entero
Definir pivoteLoc como entero
Definir aux como entero
Definir aux2 como entero
Inicio
pivote <- lalista[derecha]
pivoteLoc <- izquierda
Para i de izquierda hasta derecha con paso 1
  Si (lalista[i] <= pivote) entonces
    aux <- lalista[i]
    lalista[i] <- lalista[pivoteLoc]
    lalista[pivoteLoc] <- aux
    pivoteLoc <- pivoteLoc + 1
  FinSi
FinPara
aux2 <- lalista[derecha]
lalista[derecha] <- lalista[pivoteLoc]
lalista[pivoteLoc] <- aux2
Devolver pivoteLoc
FinFuncion
