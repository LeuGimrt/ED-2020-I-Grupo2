Pseudocódigo
Funcion binarySearch(list lalista,entero dato)

D.V.
Definir menor como entero
Definir mayor como entero
Definir centro como entero
Inicio
mayor <- tamaño(lalista) - 1
Mientras (menor <= mayor)
  centro <- (menor + mayor)/2 FinMientras
Si (lalista[centro] == dato) entonces
  Devolver centro
SiNo
  Si(lalista[centro] < dato) entonces
   menor <- centro + 1
  SiNo
   mayor <- centro - 1
  FinSi
FinSi
Devolver -1
FinFuncion


Pseudocódigo
Funcion entero linearSearch(list lalista,entero dato)

D.V.
Definir i como entero
Inicio
i <- 0
Mientras (i < tamaño(lalista) y lalista[i] < dato)
  i <- i+1
FinMientras
Si (i >= tamaño(lalista) o lalista[i] != dato) entonces
  Devolver -1
FinSi
Devolver i
FinFuncion


Pseudocódigo
Funcion entero quickSelect(list lalista,entero izquierda,entero derecha, entero k)

D.V.
Definir pivote como entero
Inicio
Si (izquierda == derecha) entonces
  Devolver lalista[izquierda]
FinSi pivote <- particion(lalista,izquierda,derecha,pivote)
Si (k = pivote) entonces
  Devolver lalista[k]
SiNo
  Si (k < pivote)
   derecha = pivote - 1
  SiNo
   izquierda = pivote + 1
  FinSi
FinSi
FinFuncion
Funcion entero particion(list lalista,entero izquierda,entero derecha)
D.V.
Definir i como entero
Definir j como entero
Definir x como arreglo
Inicio
Para j de izquierda hasta derecha con paso 1   Si (lalista[j] <= x) entonces
   swap(lalista[i], lalista[j])
   i <- i + 1
  FinSi
FinPara
swap(lalista[i], lalista[derecha])
Devolver i FinFuncion
